// 請將 index.html 內的 script 部分替換如下 const scriptURL = 'https://script.google.com/macros/s/AKfycbzG1_P4ZOEx95B1T2-3XbjEDM3VD0ALuYJbNeNCFlda5Tj3jpUc261eMdS4qr722WGV/exec'; const form = document.getElementById('enrollmentForm'); form.addEventListener('submit', e => { e.preventDefault(); const btn = document.getElementById('submitBtn'); btn.disabled = true; btn.innerText = '處理中，請稍候...'; // 收集表單資料 const formData = new FormData(form); // 處理複選框 (推薦來源) const referrals = []; document.querySelectorAll('input[name="referral"]:checked').forEach(cb => referrals.push(cb.value)); const data = Object.fromEntries(formData.entries()); data.referral = referrals.join(', '); // 合併複選項目 // 發送到 Google Sheets fetch(scriptURL, { method: 'POST', mode: 'no-cors', // 必須使用 no-cors 模式 cache: 'no-cache', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }) .then(() => { // 因為 no-cors 無法讀取回傳值，我們直接假設成功並提示 alert('報名資料已成功送出！\n系統將自動分配編號並寄送 Email 通知，請至信箱查收。'); form.reset(); btn.disabled = false; btn.innerText = '確認送出報名表'; }) .catch(error => { alert('網路連線異常，請重新嘗試。'); console.error('Error!', error.message); btn.disabled = false; btn.innerText = '確認送出報名表'; }); });
