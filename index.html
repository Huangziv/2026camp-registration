function doPost(e) { try { var params = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var timestamp = new Date(); var dateString = Utilities.formatDate(timestamp, "GMT+8", "yyyy/MM/dd HH:mm:ss"); // 生成 6 碼大寫英數亂碼 function generateRandomID(length) { var result = ''; var characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789'; for (var i = 0; i < length; i++) { result += characters.charAt(Math.floor(Math.random() * characters.length)); } return result; } var regID = "2026S-" + generateRandomID(6); // 寫入試算表 (對照 image_469060.png 欄位順序) sheet.appendRow([ dateString, // A: 時間戳記 regID, // B: 報到編號 params.email, // C: 電子郵件 params.submitter, // D: 填表人 params.studentName, // E: 學員姓名 params.gender, // F: 性別 params.school, // G: 目前就讀學校 params.grade, // H: 目前就讀年級 params.idNumber, // I: 身分證字號 params.birthday, // J: 出生年月日 params.tel, // K: 市話 params.mobile, // L: 緊急手機 params.address, // M: 地址 params.parentName, // N: 家長姓名 params.relationship, // O: 關係 params.healthStatus, // P: 健康狀況 params.frequency, // Q: 第幾次報名 params.referral, // R: 推薦來源 params.paymentStatus, // S: 繳費方式 params.referralName || "", // T: 介紹人 params.referralPhone || "", // U: 介紹人聯絡電話 params.bankLast5 || "", // V: 帳號之後五碼 params.bankDateTime || "", // W: 轉帳時間日期 params.remark, // X: 備註 regID // Y: 報到編號備份 ]); // 發送確認 Email var subject = "【報名成功】2026 科技夏令營 - " + params.studentName + " 同學"; var htmlBody = `

您好，${params.parentName} 家長：

感謝您報名「2026 未來之光：探索&科技營」。

報到編號： ${regID}

活動地點： 慈濟三重園區 (新北市三重區中正北路472號)

活動時間： 7/17 (五) - 7/18 (六)

※ 活動消息將在 三重慈少粉絲專頁 隨時發佈。

`; MailApp.sendEmail({ to: params.email, subject: subject, htmlBody: htmlBody }); return ContentService.createTextOutput(JSON.stringify({"status": "Success", "regID": regID})).setMimeType(ContentService.MimeType.JSON); } catch (e) { return ContentService.createTextOutput(JSON.stringify({"status": "Error", "message": e.message})).setMimeType(ContentService.MimeType.JSON); } }
